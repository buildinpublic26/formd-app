async function loadStats() {

  const summaryEl = document.getElementById('stats-summary');

  summaryEl.innerHTML = '<div class="loading-row" style="grid-column:span 2"><div class="spinner"></div></div>';


  const { data: sessions } = await sb

    .from('workout_sessions')

    .select('*, exercises(*)')

    .eq('user_id', currentUser.id)

    .order('session_date', { ascending: false });


  if (!sessions) return;


  // This week

  const now    = new Date();

  const monday = new Date(now); monday.setDate(now.getDate() - ((now.getDay() + 6) % 7)); monday.setHours(0,0,0,0);

  const thisWeek = sessions.filter(s => new Date(s.session_date) >= monday).length;


  // Streak

  const dateSessions = [...new Set(sessions.map(s => s.session_date))].sort((a,b)=>b.localeCompare(a));

  let streak = 0;

  for (let i = 0; i < dateSessions.length; i++) {

    const d = new Date(dateSessions[i]);

    const expected = new Date(now); expected.setDate(now.getDate() - i);

    expected.setHours(0,0,0,0);

    if (d.toDateString() === expected.toDateString()) streak++;

    else break;

  }


  // Total exercises

  const allExs = sessions.flatMap(s => s.exercises || []);

  const totalSets = allExs.reduce((a, e) => a + (e.sets || 0), 0);


  summaryEl.innerHTML = `

    <div class="stat-card"><div class="stat-val">${sessions.length}</div><div class="stat-lbl">Total sessions</div></div>

    <div class="stat-card"><div class="stat-val">${thisWeek}</div><div class="stat-lbl">This week</div></div>

    <div class="stat-card"><div class="stat-val">${streak}</div><div class="stat-lbl">Day streak</div></div>

    <div class="stat-card"><div class="stat-val">${totalSets}</div><div class="stat-lbl">Total sets</div></div>

  `;


  // Personal bests (by exercise name, highest weight_kg)

  const pbMap = {};

  allExs.forEach(e => {

    if (e.weight_kg && e.weight_kg > 0) {

      if (!pbMap[e.exercise_name] || e.weight_kg > pbMap[e.exercise_name]) {

        pbMap[e.exercise_name] = e.weight_kg;

      }

    }

  });

  const pbEntries = Object.entries(pbMap).sort((a,b) => b[1]-a[1]).slice(0,15);

  const pbCard = document.getElementById('pb-card');

  const pbList = document.getElementById('pb-list');

  if (pbEntries.length) {

    pbCard.style.display = 'block';

    pbList.innerHTML = pbEntries.map(([name, kg]) => `

      <div class="pb-row">

        <span class="pb-name">${name}</span>

        <span class="pb-weight">${kg} kg</span>

      </div>`).join('');

  }


  // Most logged exercises (frequency)

  const freqMap = {};

  allExs.forEach(e => { freqMap[e.exercise_name] = (freqMap[e.exercise_name]||0) + 1; });

  const freqEntries = Object.entries(freqMap).sort((a,b)=>b[1]-a[1]).slice(0,10);

  const freqCard = document.getElementById('freq-card');

  const freqList = document.getElementById('freq-list');

  if (freqEntries.length) {

    freqCard.style.display = 'block';

    const maxFreq = freqEntries[0][1];

    freqList.innerHTML = freqEntries.map(([name, count]) => `

      <div class="freq-bar-wrap">

        <div class="freq-bar-label">

          <span class="ex-n">${name}</span>

          <span class="ex-c">${count}×</span>

        </div>

        <div class="freq-bar-bg"><div class="freq-bar-fill" style="width:${Math.round((count/maxFreq)*100)}%"></div></div>

      </div>`).join('');

  }


  // Session type breakdown

  const typeMap = {};

  sessions.forEach(s => { typeMap[s.session_type] = (typeMap[s.session_type]||0)+1; });

  const typeEntries = Object.entries(typeMap).sort((a,b)=>b[1]-a[1]);

  const typeCard = document.getElementById('session-type-card');

  const typeList = document.getElementById('session-type-list');

  if (typeEntries.length) {

    typeCard.style.display = 'block';

    const maxType = typeEntries[0][1];

    typeList.innerHTML = typeEntries.map(([name, count]) => `

      <div class="freq-bar-wrap">

        <div class="freq-bar-label">

          <span class="ex-n">${name}</span>

          <span class="ex-c">${count} sessions</span>

        </div>

        <div class="freq-bar-bg"><div class="freq-bar-fill" style="width:${Math.round((count/maxType)*100)}%;background:var(--teal)"></div></div>

      </div>`).join('');

  }

}
